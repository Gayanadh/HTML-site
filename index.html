async function processResume() { const fileInput = document.getElementById('resumeUpload'); const updates = document.getElementById('updates').value; const statusText = document.getElementById('statusText'); if (!fileInput.files[0]) { alert("Please upload a .docx file first!"); return; } statusText.innerText = "Connecting to backend... please wait."; const formData = new FormData(); formData.append('resume', fileInput.files[0]); formData.append('updates', updates); try { // REPLACE THE URL BELOW after Step 4 const response = await fetch('https://html-site-4emx.onrender.com/upgrade', { method: 'POST', body: formData }); if (response.ok) { const blob = await response.blob(); const url = window.URL.createObjectURL(blob); const a = document.createElement('a'); a.href = url; a.download = "Upgraded_Resume.docx"; document.body.appendChild(a); a.click(); statusText.innerText = "Done! Your upgraded resume downloaded."; } } catch (error) { statusText.innerText = "Error connecting to server."; console.error(error); } }
